# file-manager
 Task is to implement File Manager using Node.js APIs.
 
 npm run start -- --username=your_username

Commands:
 .exit
 up
 rm <file>
 cd <directory>
 ls <directory>
 os commands : 
 os --EOL
 os --cpus
 os --homedir
 os --username
 os --architecture
 or 
 os --cpus --homedir --username --architecture --EOL
